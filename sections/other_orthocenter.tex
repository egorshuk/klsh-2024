    \begin{definition}\thlabel{def:isogonal}
        Точки $P, Q$ называются изогонально сопряженными, если $\angle PAB = \angle QAC$, $\angle PBC = \angle QBA$, $\angle PCB = \angle QCA$.
    \end{definition}
    \begin{theorem}\thlabel{th:O, H -- isogonal}
        Ортоцентр и центр описанной окружности изогонально сопряжены.
    \end{theorem}
\end{minipage}
\hspace{0.05\textwidth}

    \begin{lemma}\thlabel{lem:cos}
        Если $AA_1$ и $BB_1$ -- высоты в треугольнике $ABC$, то $\triangle ABC \sim \triangle A_1B_1C, \quad k = \cos \angle C$.
    \end{lemma}
