\begin{theorem}\label{th:diagonals}
    Диагонали $AC$ и $BD$ четырехугольника $ABCD$ (выпуклого или не выпуклого) перпендикулярны тогда и только тогда, когда $$AB^2 + CD^2 = BC^2 + AD^2.$$
\end{theorem}
